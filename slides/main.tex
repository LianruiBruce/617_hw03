%% Slide 2: Learning
\begin{frame}
    \frametitle{Step 1 --- Learning}

    \textbf{Research question:}

    \bigskip

    {\large Do U.S.\ states have different \alert{lifestyle profiles} based on how they use energy?}

    \bigskip
    \bigskip

    \begin{itemize}
        \item Total rankings hide \emph{how} energy is used
        \item Four sectors: Residential, Commercial, Industrial, Transportation
        \item Stakeholder: general public \& community advocates
    \end{itemize}
\end{frame}


%% Slide 3: Winnowing — Data
\begin{frame}
    \frametitle{Step 2 --- Winnowing: Data Source}

    \begin{block}{Source}
        CORGIS / U.S.\ EIA SEDS --- 51 states, 2010--2019
    \end{block}

    \bigskip

    \textbf{Columns used per sector:}
    \begin{itemize}
        \item Coal, Natural Gas, Petroleum, Wood, Geothermal, \ldots
        \item Summed into 4 sector totals, then normalized to \textbf{\% shares}
    \end{itemize}

    \bigskip

    \textbf{Why necessary \& sufficient:}
    \begin{itemize}
        \item State + Year identify each row
        \item 4 shares define the profile; \% removes state-size effect
    \end{itemize}
\end{frame}


%% Slide 4: Winnowing — Limitations
\begin{frame}
    \frametitle{Step 2 --- Winnowing: Limitations}

    \bigskip

    \begin{itemize}
        \item \alert{Direct-fuel only} --- grid electricity is excluded

        \bigskip
        \item States that heat with electricity (e.g.\ Florida) appear lower in Residential share

        \bigskip
        \item Missing or very small values recorded as 0 in source data
    \end{itemize}
\end{frame}
